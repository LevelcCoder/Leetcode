//假设不同的节点
class Solution {
public:
    int minFlipsMonoIncr(string s) {
        int one_cnt[100010] = {0};
        int n = s.size();
        for(int i=0;i<n;i++){
            if(i>0) one_cnt[i] = one_cnt[i-1];
            if(s[i]=='1')
            one_cnt[i]++;

        }
        int cnt = one_cnt[n-1];
        int ans = n - cnt;
        for(int i=0;i<n;i++){
            ans = min(ans,one_cnt[i]+n-i-1-(cnt-one_cnt[i]));
        }
        return ans;
    }
};
