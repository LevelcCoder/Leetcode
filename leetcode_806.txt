//c++       easy
class Solution {
public:
    vector<int> numberOfLines(vector<int>& widths, string s) {
        int rows = 0;
        int sum = 0;
        for(int i=0;i<s.size();i++){
            
            sum+=widths[s[i]-'a'];
            if(sum>100){
                rows++;
                sum = 0;
                sum+=widths[s[i]-'a'];
            }
        }
        if(sum==0)
        return {rows,sum};
        else{
            return {rows+1,sum};
        }
    }
};
