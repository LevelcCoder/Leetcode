class Solution {
public:
    int binaryGap(int n) {
        int low_pos = -1;
        int len = 0;
        int ans = 0;
        while(n){
            if(n%2==1){
                if(low_pos>-1){
                    ans = max(ans,len-low_pos);
                }

                low_pos = len;
            }
            len++;
            
            n/=2;
        }
        return ans;
    }
};
